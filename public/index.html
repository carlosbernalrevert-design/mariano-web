const form = document.getElementById('marIAnoForm'); const input = document.getElementById('newsUrl'); const submitBtn = document.getElementById('submitBtn'); const resultBox = document.getElementById('resultBox'); const resultHeader = document.getElementById('resultHeader'); const resultContent = document.getElementById('resultContent'); const errorBox = document.getElementById('errorBox'); form.addEventListener('submit', async (e) => { e.preventDefault(); const url = input.value.trim(); if (!url) return; // Reset de estados errorBox.style.display = 'none'; resultBox.style.display = 'none'; submitBtn.disabled = true; submitBtn.textContent = 'pensando en obviedades'; try { const response = await fetch('/transformar', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ url }) }); const data = await response.json(); if (!response.ok) { throw new Error(data.error || 'Ocurrió un error inesperado.'); } resultHeader.textContent = data.header || ''; resultContent.textContent = data.content || ''; resultBox.style.display = 'block'; } catch (err) { errorBox.textContent = err.message; errorBox.style.display = 'block'; } finally { submitBtn.disabled = false; submitBtn.textContent = 'MarlAnizar'; } });
